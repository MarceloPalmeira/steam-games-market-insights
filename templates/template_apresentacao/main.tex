\begin{frame}{Variável-alvo: \texttt{success\_commercial}}
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.47\textwidth}
    \begin{block}{Definição}
      Proxy binária de sucesso comercial estimado, \textbf{não receita real}.
    \end{block}
    \small
    Top 30\% do score composto por:
    \begin{itemize}
      \item owners estimados;
      \item total de reviews;
      \item peak CCU, recomendações e tempo médio.
    \end{itemize}
    Componentes do alvo e \texttt{positive\_ratio} foram removidos das features.
  \end{column}

  \end{column}
\end{columns}

\begin{frame}{Evidências principais da EDA}
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.43\textwidth}
    \small
    \begin{itemize}
      \item Multiplayer: 44,9\% de sucesso estimado vs. 26,7\% nos demais.
      \item Pagos: 33,4\% vs. gratuitos: 17,6\%.
      \item Testes Mann-Whitney e qui-quadrado: \(p < 10^{-300}\).
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Leitura:} associação estatística no conjunto analisado, não causalidade.
  \end{column}

\begin{frame}{Insights acionáveis}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.32\textwidth}
    \begin{block}{Produto social}
      Multiplayer está associado a maior taxa de sucesso estimado.\newline
      \textbf{Ação:} avaliar co-op, competitivo ou eventos online quando fizer sentido ao gênero.
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{Preço}
      Gratuito não apresentou maior sucesso estimado no conjunto analisado.\newline
      \textbf{Ação:} testar preço baixo, demo ou promoção sem depender automaticamente de free-to-play.
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{Engajamento}
      Achievements aparecem associados a maior taxa de sucesso estimado.\newline
      \textbf{Ação:} conectar conquistas a progresso real de gameplay.
    \end{block}
  \end{column}
\end{columns}
\vspace{0.2cm}
\centering
\footnotesize Avaliar sucesso das ações por reviews, peak CCU, retenção, conversão e tempo médio após o lançamento.
\end{frame}
